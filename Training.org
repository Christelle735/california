* Training

#+begin_src jupyter-python
rf = RandomForestRegressor(random_state=42)

param_grid = { 
    'n_estimators': [200, 500],
    'max_features': ['auto', 'sqrt', 'log2'],
    'max_depth' : [4,5,6,7,8]
}

gm = GridSearchCV(estimator=rf, param_grid=param_grid, cv= 5, oob_score=True)
gm.fit(X, y)
gm.best_params_
#+end_src
